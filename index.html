// ════════════════════════════════════════
// نظام تسليم واستلام الأجهزة — Apps Script
// ════════════════════════════════════════
const HANDOVER_SHEET = 'العهد';
const EMP_SHEET      = 'الموظفون';
const DRIVE_FOLDER   = 'عهد_الأجهزة_صور';
const COLS = [
  'رقم_الوثيقة','التاريخ','الوقت','الموقع',
  'نوع_الجهاز','الرقم_التسلسلي',
  'رقم_موظف_المسلم_له','اسم_موظف_المسلم_له',
  'ملاحظات_التسليم','الحالة',
  'تاريخ_الاستلام','رقم_موظف_المستلم','اسم_موظف_المستلم','ملاحظات_الاستلام',
  'صورة_الأمام','صورة_الخلف'
];

function doGet(e)  { return handle(e); }
function doPost(e) { return handle(e); }

function handle(e) {
  const p = e.parameter || {};
  const b = e.postData ? JSON.parse(e.postData.contents) : {};
  const action = p.action || b.action;
  try {
    if (action === 'init')        return initSheet();
    if (action === 'getAll')      return getAll();
    if (action === 'add')         return addRecord(b);
    if (action === 'receive')     return markReceived(b);
    if (action === 'getEmployee') return getEmployee(p.id);
    if (action === 'getDeviceTypes') return getDeviceTypes();
    if (action === 'uploadPhoto') return uploadPhoto(b);
    return out({error: 'unknown action'});
  } catch(err) {
    return out({error: err.message});
  }
}

function initSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(HANDOVER_SHEET);
  if (!sh) sh = ss.insertSheet(HANDOVER_SHEET);
  if (sh.getLastRow() === 0) {
    sh.appendRow(COLS);
    sh.getRange(1,1,1,COLS.length)
      .setBackground('#0B1D3A').setFontColor('#FFFFFF')
      .setFontWeight('bold').setHorizontalAlignment('center');
    sh.setFrozenRows(1);
    [1,2,3,10].forEach(c => sh.setColumnWidth(c, 150));
  }
  return out({success: true});
}

function getAll() {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HANDOVER_SHEET);
  if (!sh || sh.getLastRow() < 2) return out({data: []});
  const rows = sh.getDataRange().getValues();
  const headers = rows[0];
  const data = rows.slice(1).map((r,i) => {
    const obj = {}; headers.forEach((h,j) => obj[h] = r[j]);
    obj._row = i + 2; return obj;
  });
  return out({data});
}

function addRecord(d) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HANDOVER_SHEET);
  if (!sh) return out({error: 'Sheet not found'});
  sh.appendRow([
    d['رقم_الوثيقة'], d['التاريخ'], d['الوقت'], d['الموقع'],
    d['نوع_الجهاز'], d['الرقم_التسلسلي']||'',
    d['رقم_موظف_المسلم_له'], d['اسم_موظف_المسلم_له'],
    d['ملاحظات_التسليم']||'', 'معلقة',
    '', '', '', '', '', ''
  ]);
  return out({success: true});
}

function markReceived(d) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HANDOVER_SHEET);
  const rows = sh.getDataRange().getValues();
  const headers = rows[0];
  const idx = r => headers.indexOf(r) + 1;
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(d['رقم_الوثيقة'])) {
      const now = new Date().toLocaleString('ar-SA');
      sh.getRange(i+1, idx('الحالة')).setValue('مُستلمة');
      sh.getRange(i+1, idx('تاريخ_الاستلام')).setValue(d['تاريخ_الاستلام']||now);
      sh.getRange(i+1, idx('رقم_موظف_المستلم')).setValue(d['رقم_موظف_المستلم']||'');
      sh.getRange(i+1, idx('اسم_موظف_المستلم')).setValue(d['اسم_موظف_المستلم']||'');
      sh.getRange(i+1, idx('ملاحظات_الاستلام')).setValue(d['ملاحظات_الاستلام']||'');
      return out({success: true});
    }
  }
  return out({error: 'record not found'});
}

function getEmployee(id) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(EMP_SHEET);
  if (!sh) return out({name: null});
  const rows = sh.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]).trim() === String(id).trim())
      return out({name: String(rows[i][1])});
  }
  return out({name: null});
}

function getDeviceTypes() {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('أنواع_العهد');
  if (!sh || sh.getLastRow() < 2) return out({types: []});
  const rows = sh.getDataRange().getValues();
  // First column, skip header row
  const types = rows.slice(1).map(r => String(r[0]).trim()).filter(Boolean);
  return out({types});
}

function uploadPhoto(d) {
  // d: { docId, side, base64, mimeType }
  let folder;
  const folders = DriveApp.getFoldersByName(DRIVE_FOLDER);
  folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(DRIVE_FOLDER);

  const blob = Utilities.newBlob(
    Utilities.base64Decode(d.base64),
    d.mimeType || 'image/jpeg',
    `${d.docId}_${d.side}.jpg`
  );
  const file = folder.createFile(blob);
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  const url = 'https://drive.google.com/thumbnail?id=' + file.getId() + '&sz=w800';

  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HANDOVER_SHEET);
  const rows = sh.getDataRange().getValues();
  const headers = rows[0];
  const col = headers.indexOf(d.side === 'front' ? 'صورة_الأمام' : 'صورة_الخلف') + 1;
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(d.docId)) {
      sh.getRange(i+1, col).setValue(url);
      return out({success: true, url});
    }
  }
  return out({error: 'record not found'});
}

function out(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
